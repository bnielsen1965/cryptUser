 71:      * Load this user defined by this object from the data source
 72:      * @return boolean TRUE on success, FALSE on failure
 73:      */
 74:     private function loadUser() {
 75:         // retrieve user's data from the data source
 76:         $userData = $this->dataSource->getUserByName($this->username);
 77:         
 78:         // if user data returned then process
 79:         if ($userData) {
 80:             // use data values for this user
 81:             $this->username = $userData['username'];
 82:             $this->passwordHash = $userData['passwordHash'];
 83:             $this->flags = $userData['flags'];
 84:             $this->sslKey = $userData['sslKey'];
 85: 
 86:             // authenticate user by checking data source password hash with a hash of the provided password
 87:             if ($this->passwordHash == $this->hashPassword($this->password, $this->passwordHash)) {
 88:                 // authentication passed
 89:                 $this->authenticated = TRUE;
 90:                 
 91:                 // set the user's primary SSL key
 92:                 $this->setPrimaryKey(SSLKey::parsePrivateKey($this->sslKey), SSLKey::parseCertificate($this->sslKey));
 93:             }
 94:             else {
 95:                 // authentication failed
 96:                 $this->authenticated = FALSE;
 97:             }
 98:             
 99:             // load successful
100:             return TRUE;
101:         }
102:         
103:         // failed to load the user
104:         return FALSE;
105:     }

248:     /**
249:      * Change the user's password and the SSL Key elements that rely on the password.
250:      * @param string $password The new password to use.
251:      * @param string $callback Optional callback function used by the application for
252:      * post password change maintenance like re-encryption of data. The callback must
253:      * accept two CryptUser objects, the first object is the CryptUser with the new
254:      * password and encryption key, the second object is the Cryptuser with the old
255:      * password and encryption key.
256:      * @return boolean TRUE on success, FALSE on failure.
257:      */
258:     public function changePassword($password, $callback = NULL) {
259:         // if a callback is provided then prepare to use callback
260:         if (!empty($callback)) {
261:             // if this user is not authenticated then decryption will not be possible so we fail
262:             if (!$this->isAuthenticated()) return FALSE;
263:             
264:             // create a clone of this user to be sent to callback
265:             $oldCryptUser = clone $this;
266:         }
267:         
268:         $this->password = $password;
269:         $this->passwordHash = $this->hashPassword($password);
270:         
271:         // set a new primary key
272:         $this->setPrimaryKey();
273:         
274:         // set the sslKey value using the new primary key
275:         $this->sslKey = $this->primaryKey->getPrivateKey() . $this->primaryKey->getCertificate();
276:         
277:         // if a callback is provided then call now with the old and this new user
278:         if (!empty($callback)) call_user_func($callback, $oldCryptUser, $this);
279:         
280:         // password change complete
281:         return TRUE;
282:     }

327:         // if password empty then return empty
328:         if( empty($password) ) return '';
329: 
330:         if( is_null($salt) ) $salt = cryptUser::bestSalt();
331: 
332:         return crypt($password, $salt);

340:     public static function bestSalt() {
341:         if( defined('CRYPT_SHA512') && CRYPT_SHA512 == 1 ) return '$6$' . SSLKey::makePhrase(16) . '$';
342:         if( defined('CRYPT_SHA256') && CRYPT_SHA256 == 1 ) return '$5$' . SSLKey::makePhrase(16) . '$';
343:         if( defined('CRYPT_BLOWFISH') && CRYPT_BLOWFISH == 1 ) return '$2a$07$' . base64_encode(SSLKey::makePhrase(22)) . '$';
344:         if( defined('CRYPT_MD5') && CRYPT_MD5 == 1 ) return '$1$' . SSLKey::makePhrase(12) . '$';
345:         if( defined('CRYPT_EXT_DES') && CRYPT_EXT_DES == 1 ) return '_' . SSLKey::makePhrase(8);
346:         if( defined('CRYPT_STD_DES') && CRYPT_STD_DES == 1 ) return SSLKey::makePhrase(2);
347:         return '';
