 25: /**
 26:  * A MySQL based data source object for the CryptUsers.
 27:  *
 28:  * @author Bryan Nielsen, bnielsen1965@gmail.com

 31: class CryptMySQLSource implements CryptDataSource {
 32:     private $mysqli;
 33:     private $usersTable;
 34:     
 35:     
 36:     /**
 37:      * Constructor to set up the data source
 38:      * 
 39:      */
 40:     public function __construct($databaseConfig) {
 41:         // connect if parameters provided
 42:         if (!empty($databaseConfig['host']) && !empty($databaseConfig['username']) && !empty($databaseConfig['password']) && !empty($databaseConfig['database'])) {
 43:             $this->mysqli = new mysqli($databaseConfig['host'], $databaseConfig['username'], $databaseConfig['password'], $databaseConfig['database']);
 44:             
 45:             if ($this->mysqli->connect_errno) {
 46:                 throw new Exception("Failed to connect to MySQL: (" . $this->mysqli->connect_errno . ") " . $this->mysqli->connect_error);
 47:             }
 48:         }
 49:         
 50:         if (!empty($databaseConfig['usersTable'])) $this->usersTable = $databaseConfig['usersTable'];
 51:         else $this->usersTable = 'users';
 52:     }
 53:     
 54:     
 55:     /**
 56:      * Get source type
 57:      * 
 58:      * @return string The data source type.
 59:      */
 60:     public function getSourceType() {
 61:         return 'MySQL';
 62:     }
 63:     
 64:     
 65:     /**
 66:      * Get array of users that match a given name.
 67:      * @param string $username The username to search for in the data source.
 68:      * @return array|boolean An array of arrays containing user elements to create a user
 69:      * or FALSE if not found.
 70:      */
 71:     public function getUserByName($username) {
 72:         $sql = "SELECT * FROM " . $this->usersTable . " WHERE username='" . $this->mysqli->real_escape_string($username) . "'";
 73:         $rs = $this->mysqli->query($sql);
 74:         if ($rs && $rs->num_rows) {
 75:             return $rs->fetch_assoc();
 76:         }
 77:         
 78:         // failed to find user
 79:         return FALSE;
 80:     }
 81:     
 82:     
 83:     /**
 84:      * Get a list of usernames
 85:      * @return array An array of strings containing the usernames from the data source.
 86:      */
 87:     public function getUsernames() {
 88:         $sql = "SELECT * FROM " . $this->usersTable;
 89:         $rs = $this->mysqli->query($sql);
 90:         if ($rs && $rs->num_rows) {
 91:             $usernames = array();
 92:             while ($row = $rs->fetch_assoc()) {
 93:                 $usernames[] = $row['username'];
 94:             }
 95:             return $usernames;
 96:         }
 97:         
 98:         // failed to find user
 99:         return FALSE;
100:     }

102:     
103:     /**
104:      * Save the provided user details in the data source.
105:      * @param array $user An array of user elements to be saved.
106:      * @return boolean Returns TRUE on success and FALSE on failure.
107:      */
108:     public function saveUser($user) {
109:         if ($this->getUserByName($user['username']) !== FALSE) {
110:             // user exists, update
111:             $sql = "UPDATE " . $this->usersTable . " SET " .
112:                     "passwordHash='" . $this->mysqli->real_escape_string($user['passwordHash']) . "', " .
113:                     "sslKey='" . $this->mysqli->real_escape_string($user['sslKey']) . "', " .
114:                     "flags='" . $this->mysqli->real_escape_string($user['flags']) . "' " .
115:                     "WHERE username='" . $this->mysqli->real_escape_string($user['username']) . "'";
116:         }
117:         else {
118:             // user does not exist, insert
119:             $sql = "INSERT INTO " . $this->usersTable . "(username, passwordHash, sslKey, flags) VALUES (" .
120:                 "'" . $this->mysqli->real_escape_string($user['username']) . "', " .
121:                 "'" . $this->mysqli->real_escape_string($user['passwordHash']) . "', " .
122:                 "'" . $this->mysqli->real_escape_string($user['sslKey']) . "', " .
123:                 "'" . $this->mysqli->real_escape_string($user['flags']) . "'" .
124:                 ")";
125:         }
126:         
127:         return $this->mysqli->query($sql);
128:     }

133:      * @param string $username The username of the user to delete.
134:      * @return boolean Returns TRUE on success and FALSE on failure.
135:      */
136:     public function deleteUser($username) {
137:         $sql = "DELETE FROM " . $this->usersTable . " WHERE username='" . $this->mysqli->real_escape_string($username) . "'";

144:      * @param array $users An array of user rows to search.
145:      * @return integer|boolean The array index for the user name or FALSE if not found.
146:      */
147:     private function searchUsersForUser($users, $username) {
148:         if ($users) {
